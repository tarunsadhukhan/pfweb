<script>
$(document).ready(function() {
    $('#mcno').on('change', function() {
        var mcno = $('#mcno').val();
        var rundate = $('#rundate').val();
        var runshift = $('#runshift').val();
        var company_id = $('#company_id').val();

        $.ajax({
            url: "<?php echo base_url('controller_name/fetch_data'); ?>",
            type: "POST",
            data: {mcno: mcno, rundate: rundate, runshift: runshift, company_id: company_id},
            dataType: "json",
            success: function(response) {
                if (response.success) {
                    // Process and display the fetched data
                    var name = response.name;
                    var weight = response.weight;

                    $('#resultContainer').html('Name: ' + name + '<br>Weight: ' + weight);
                } else {
                    $('#resultContainer').text('No data found for the given parameters.');
                }
            }
        });
    });
});
</script>
